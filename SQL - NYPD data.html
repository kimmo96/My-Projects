# Define the SQL query to find noise complaints
query = '''

        WITH noise_complaints AS (
          SELECT year, agency, complaint_type, num_complaints,
                SUM(num_complaints) OVER (PARTITION BY year, agency) AS total_complaints
          FROM df2
          WHERE complaint_type LIKE '%Noise%'
        )
        SELECT *
        FROM noise_complaints
        WHERE total_complaints < 10000
        ORDER BY total_complaints DESC

# Create a figure and axis
fig, ax = plt.subplots()

# Set the width for each bar
bar_width = 0.7

# Define the colors for the bars
colors = ['#1f77b4', '#ff7f0e', '#2ca02c', '#d62728', '#9467bd', '#8c564b', '#e377c2', '#7f7f7f', '#bcbd22', '#17becf', '#aec7e8']
# Dictionary to track complaint types and assigned colors
complaint_colors = {}

# Iterate over each year
for i, year in enumerate(complaints['year'].unique()):
    # Get the data for the current year
    year_data = complaints[complaints['year'] == year]

    # Calculate the x-axis positions for the bars
    x = [i * 5 + j * bar_width for j in range(5)]

    # Plot the bars for the current year
    for j, complaint_type in enumerate(year_data['complaint_type'].unique()):
        complaint_type_data = year_data[year_data['complaint_type'] == complaint_type]

        # Check if the complaint type has been assigned a color before
        if complaint_type in complaint_colors:
            color = complaint_colors[complaint_type]
        else:
            # Assign a color to the complaint type
            color = colors[len(complaint_colors) % len(colors)]
            complaint_colors[complaint_type] = color

        ax.bar(x[j], complaint_type_data['num_complaints'], width=bar_width, color=color)

# Set the x-axis labels
x_ticks = [i * 5 + 2.5 for i in range(len(complaints['year'].unique()))]
ax.set_xticks(x_ticks)
ax.set_xticklabels(complaints['year'].unique(), ha='right')

# Set the y-axis label
ax.set_ylabel('Number of Complaints')
ax.set_xlabel('Year')
# Set the title
ax.set_title('5 Most Common Number of Complaints by Year')

# Create custom legend
legend_elements = [plt.Rectangle((0, 0), 1, 1, color=complaint_colors[complaint_type], label=complaint_type) for complaint_type in complaint_colors]
ax.legend(handles=legend_elements)

# Show the plot
plt.show()

# 4b)
# Run SQL query and display results

query = '''
SELECT
    incident_zip AS zipcode,
    COUNT(*) AS total_complaints,
    SUM(IF(
        EXTRACT(DAYOFWEEK FROM created_date) = 7 AND
        EXTRACT(DAYOFWEEK FROM closed_date) = 2 AND
        TIMESTAMP_TRUNC(closed_date, DAY) <= TIMESTAMP_TRUNC(TIMESTAMP_ADD(created_date, INTERVAL 1 DAY), DAY),
        1,
        0
    )) AS closed_in_1_day,
    COUNT(*) - SUM(IF(
        EXTRACT(DAYOFWEEK FROM created_date) = 7 AND
        EXTRACT(DAYOFWEEK FROM closed_date) = 2 AND
        TIMESTAMP_TRUNC(closed_date, DAY) <= TIMESTAMP_TRUNC(TIMESTAMP_ADD(created_date, INTERVAL 1 DAY), DAY),
        1,
        0
    )) AS closed_in_more_than_1_day
FROM
    `bigquery-public-data.new_york.311_service_requests`
WHERE
    incident_zip IS NOT NULL
GROUP BY
    zipcode
ORDER BY
    total_complaints DESC

'''

df3 = client.query(query).to_dataframe()
df3.head(5)

This query analyzes service requests data and provides information on the number of complaints, the number of complaints closed within 1 day, and the number of complaints closed in more than 1 day, grouped by zip code. The results are ordered by the total number of complaints in descending order.

# Execute the query to find the zip code with the highest ratio of complaints per population (closed within 1 business day)
query = '''

        SELECT
            z.zipcode, SUM(z.population) AS total_population,
            t.closed_in_1_day, t.total_complaints,
            (t.closed_in_1_day / SUM(z.population)) AS complaints_population_ratio
        FROM
            zip_df z
        JOIN
            df3 t ON z.zipcode = t.zipcode
        WHERE
             t.closed_in_1_day IS NOT NULL
        GROUP BY
             z.zipcode
        ORDER BY
              complaints_population_ratio DESC
        LIMIT 1;

'''
